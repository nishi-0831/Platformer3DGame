461 // 円錐の底面の半径
462 // MEMO: tanθ = 対辺 / 底辺 : 対辺 = 底辺 * tanθ
463 float baseRadius = _distance * std::tanf(DirectX::XMConvertToRadians(_fovAngleDegree / 2.0f));
464
465 // 上下方向を決定
466 Vector3 up = std::fabsf(DirectX::XMVectorGetY(forward)) < 0.9f
467 ? Vector3{ 0.0f,1.0f,0.0f }
468 : Vector3{ 1.0f,0.0f,0.0f };
469
470 // 右方向を計算

472
