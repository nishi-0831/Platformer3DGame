38 static bool SequencerAddDelButton(ImDrawList* draw_list, ImVec2 pos, bool add = true)
39 {
40 ImGuiIO& io = ImGui::GetIO();
41 ImRect btnRect(pos, ImVec2(pos.x + 16, pos.y + 16));
42 bool overBtn = btnRect.Contains(io.MousePos);
43 bool containedClick = overBtn && btnRect.Contains(io.MouseClickedPos[0]);
44 bool clickedBtn = containedClick && io.MouseReleased[0];
45 int btnColor = overBtn ? 0xAAEAFFAA : 0x77A3B2AA;
46 if (containedClick && io.MouseDownDuration[0] > 0)
47 btnRect.Expand(2.0f);
48
49 float midy = pos.y + 16 / 2 - 0.5f;
50 float midx = pos.x + 16 / 2 - 0.5f;
51 draw_list->AddRect(btnRect.Min, btnRect.Max, btnColor, 4);
52 draw_list->AddLine(ImVec2(btnRect.Min.x + 3, midy), ImVec2(btnRect.Max.x - 3, midy), btnColor, 2);
53 if (add)
54 draw_list->AddLine(ImVec2(midx, btnRect.Min.y + 3), ImVec2(midx, btnRect.Max.y - 3), btnColor, 2);
55 return clickedBtn;
56 }
