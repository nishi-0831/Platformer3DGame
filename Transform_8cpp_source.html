39void mtgb::Transform::Compute()
40{
41 using DirectX::XMMatrixTranslation;
42 using DirectX::XMMatrixRotationQuaternion;
43 using DirectX::XMQuaternionNormalize;
44 using DirectX::XMConvertToRadians;
45 using DirectX::XMMatrixScaling;
46
47 // 親の行列を取得
48 Matrix4x4 parentMat{};
49 GenerateParentMatrix(&parentMat);
50
51
52 // 現在のワールド座標から行列を作成
53 Matrix4x4 currWorldMat =
54 XMMatrixScaling(scale.x, scale.y, scale.z) *
55 XMMatrixRotationQuaternion(XMQuaternionNormalize(rotate)) *
56 XMMatrixTranslation(position.x, position.y, position.z);
57 Matrix4x4 parentInverse = DirectX::XMMatrixInverse(nullptr, prevParentMatrix_);
58
59 matrixLocal_ = currWorldMat * parentInverse;
60 DecomposeMatrixImpl(&localPosition_, &localRotate_, &localScale_, matrixLocal_);
61
62 matrixTranslate_ = XMMatrixTranslation(localPosition_.x, localPosition_.y, localPosition_.z);
63 matrixRotate_ = XMMatrixRotationQuaternion(
64 XMQuaternionNormalize(localRotate_));
65 matrixScale_ = XMMatrixScaling(localScale_.x, localScale_.y, localScale_.z);
66
67 GenerateWorldMatrix(&matrixWorld_);
68 // ワールド行列を計算
69 GenerateWorldRotationMatrix(&matrixWorldRot_); // ワールド回転行列更新
70
71 // ワールド行列からワールド座標系の値を逆算して更新
72 DecomposeMatrixImpl(&position, &rotate, &scale, matrixWorld_);
73
74 prevParentMatrix_ = parentMat;
75}
