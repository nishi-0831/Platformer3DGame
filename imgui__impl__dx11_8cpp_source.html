110static void ImGui_ImplDX11_SetupRenderState(ImDrawData* draw_data, ID3D11DeviceContext* device_ctx)
111{
112 ImGui_ImplDX11_Data* bd = ImGui_ImplDX11_GetBackendData();
113
114 // Setup viewport
115 D3D11_VIEWPORT vp = {};
116 vp.Width = draw_data->DisplaySize.x * draw_data->FramebufferScale.x;
117 vp.Height = draw_data->DisplaySize.y * draw_data->FramebufferScale.y;
118 vp.MinDepth = 0.0f;
119 vp.MaxDepth = 1.0f;
120 vp.TopLeftX = vp.TopLeftY = 0;
121 device_ctx->RSSetViewports(1, &vp);
122
123 // Setup orthographic projection matrix into our constant buffer
124 // Our visible imgui space lies from draw_data->DisplayPos (top left) to draw_data->DisplayPos+data_data->DisplaySize (bottom right). DisplayPos is (0,0) for single viewport apps.
125 D3D11_MAPPED_SUBRESOURCE mapped_resource;
126 if (device_ctx->Map(bd->pVertexConstantBuffer, 0, D3D11_MAP_WRITE_DISCARD, 0, &mapped_resource) == S_OK)
127 {
128 VERTEX_CONSTANT_BUFFER_DX11* constant_buffer = (VERTEX_CONSTANT_BUFFER_DX11*)mapped_resource.pData;
129 float L = draw_data->DisplayPos.x;
130 float R = draw_data->DisplayPos.x + draw_data->DisplaySize.x;
131 float T = draw_data->DisplayPos.y;
132 float B = draw_data->DisplayPos.y + draw_data->DisplaySize.y;
133 float mvp[4][4] =
134 {
135 { 2.0f/(R-L), 0.0f, 0.0f, 0.0f },
136 { 0.0f, 2.0f/(T-B), 0.0f, 0.0f },
137 { 0.0f, 0.0f, 0.5f, 0.0f },
138 { (R+L)/(L-R), (T+B)/(B-T), 0.5f, 1.0f },
139 };
140 memcpy(&constant_buffer->mvp, mvp, sizeof(mvp));
141 device_ctx->Unmap(bd->pVertexConstantBuffer, 0);
142 }
143
144 // Setup shader and vertex buffers
145 unsigned int stride = sizeof(ImDrawVert);
146 unsigned int offset = 0;
147 device_ctx->IASetInputLayout(bd->pInputLayout);
148 device_ctx->IASetVertexBuffers(0, 1, &bd->pVB, &stride, &offset);
149 device_ctx->IASetIndexBuffer(bd->pIB, sizeof(ImDrawIdx) == 2 ? DXGI_FORMAT_R16_UINT : DXGI_FORMAT_R32_UINT, 0);
150 device_ctx->IASetPrimitiveTopology(D3D11_PRIMITIVE_TOPOLOGY_TRIANGLELIST);
151 device_ctx->VSSetShader(bd->pVertexShader, nullptr, 0);
152 device_ctx->VSSetConstantBuffers(0, 1, &bd->pVertexConstantBuffer);
153 device_ctx->PSSetShader(bd->pPixelShader, nullptr, 0);
154 device_ctx->PSSetSamplers(0, 1, &bd->pFontSampler);
155 device_ctx->GSSetShader(nullptr, nullptr, 0);
156 device_ctx->HSSetShader(nullptr, nullptr, 0); // In theory we should backup and restore this as well.. very infrequently used..
157 device_ctx->DSSetShader(nullptr, nullptr, 0); // In theory we should backup and restore this as well.. very infrequently used..
158 device_ctx->CSSetShader(nullptr, nullptr, 0); // In theory we should backup and restore this as well.. very infrequently used..
159
160 // Setup render state
161 const float blend_factor[4] = { 0.f, 0.f, 0.f, 0.f };
162 device_ctx->OMSetBlendState(bd->pBlendState, blend_factor, 0xffffffff);
163 device_ctx->OMSetDepthStencilState(bd->pDepthStencilState, 0);
164 device_ctx->RSSetState(bd->pRasterizerState);
165}

167// Render function
168void ImGui_ImplDX11_RenderDrawData(ImDrawData* draw_data)
169{
170 // Avoid rendering when minimized
171 if (draw_data->DisplaySize.x <= 0.0f || draw_data->DisplaySize.y <= 0.0f)
172 return;
173
174 ImGui_ImplDX11_Data* bd = ImGui_ImplDX11_GetBackendData();
175 ID3D11DeviceContext* device = bd->pd3dDeviceContext;
176
177 // Catch up with texture updates. Most of the times, the list will have 1 element with an OK status, aka nothing to do.
178 // (This almost always points to ImGui::GetPlatformIO().Textures[] but is part of ImDrawData to allow overriding or disabling texture updates).
179 if (draw_data->Textures != nullptr)
180 for (ImTextureData* tex : *draw_data->Textures)
181 if (tex->Status != ImTextureStatus_OK)
182 ImGui_ImplDX11_UpdateTexture(tex);
183
184 // Create and grow vertex/index buffers if needed
185 if (!bd->pVB || bd->VertexBufferSize < draw_data->TotalVtxCount)
186 {
187 if (bd->pVB) { bd->pVB->Release(); bd->pVB = nullptr; }
188 bd->VertexBufferSize = draw_data->TotalVtxCount + 5000;
189 D3D11_BUFFER_DESC desc = {};
190 desc.Usage = D3D11_USAGE_DYNAMIC;
191 desc.ByteWidth = bd->VertexBufferSize * sizeof(ImDrawVert);
192 desc.BindFlags = D3D11_BIND_VERTEX_BUFFER;
193 desc.CPUAccessFlags = D3D11_CPU_ACCESS_WRITE;
194 desc.MiscFlags = 0;
195 if (bd->pd3dDevice->CreateBuffer(&desc, nullptr, &bd->pVB) < 0)
196 return;
197 }
198 if (!bd->pIB || bd->IndexBufferSize < draw_data->TotalIdxCount)
199 {
200 if (bd->pIB) { bd->pIB->Release(); bd->pIB = nullptr; }
201 bd->IndexBufferSize = draw_data->TotalIdxCount + 10000;
202 D3D11_BUFFER_DESC desc = {};
203 desc.Usage = D3D11_USAGE_DYNAMIC;
204 desc.ByteWidth = bd->IndexBufferSize * sizeof(ImDrawIdx);
205 desc.BindFlags = D3D11_BIND_INDEX_BUFFER;
206 desc.CPUAccessFlags = D3D11_CPU_ACCESS_WRITE;
207 if (bd->pd3dDevice->CreateBuffer(&desc, nullptr, &bd->pIB) < 0)
208 return;
209 }
210
211 // Upload vertex/index data into a single contiguous GPU buffer
212 D3D11_MAPPED_SUBRESOURCE vtx_resource, idx_resource;
213 if (device->Map(bd->pVB, 0, D3D11_MAP_WRITE_DISCARD, 0, &vtx_resource) != S_OK)
214 return;
215 if (device->Map(bd->pIB, 0, D3D11_MAP_WRITE_DISCARD, 0, &idx_resource) != S_OK)
216 return;
217 ImDrawVert* vtx_dst = (ImDrawVert*)vtx_resource.pData;
218 ImDrawIdx* idx_dst = (ImDrawIdx*)idx_resource.pData;
219 for (int n = 0; n < draw_data->CmdListsCount; n++)
220 {
221 const ImDrawList* draw_list = draw_data->CmdLists[n];
222 memcpy(vtx_dst, draw_list->VtxBuffer.Data, draw_list->VtxBuffer.Size * sizeof(ImDrawVert));
223 memcpy(idx_dst, draw_list->IdxBuffer.Data, draw_list->IdxBuffer.Size * sizeof(ImDrawIdx));
224 vtx_dst += draw_list->VtxBuffer.Size;
225 idx_dst += draw_list->IdxBuffer.Size;
226 }
227 device->Unmap(bd->pVB, 0);
228 device->Unmap(bd->pIB, 0);

230 // Backup DX state that will be modified to restore it afterwards (unfortunately this is very ugly looking and verbose. Close your eyes!)
231 struct BACKUP_DX11_STATE
232 {
233 UINT ScissorRectsCount, ViewportsCount;
234 D3D11_RECT ScissorRects[D3D11_VIEWPORT_AND_SCISSORRECT_OBJECT_COUNT_PER_PIPELINE];
235 D3D11_VIEWPORT Viewports[D3D11_VIEWPORT_AND_SCISSORRECT_OBJECT_COUNT_PER_PIPELINE];
236 ID3D11RasterizerState* RS;
237 ID3D11BlendState* BlendState;
238 FLOAT BlendFactor[4];
239 UINT SampleMask;
240 UINT StencilRef;
241 ID3D11DepthStencilState* DepthStencilState;
242 ID3D11ShaderResourceView* PSShaderResource;
243 ID3D11SamplerState* PSSampler;
244 ID3D11PixelShader* PS;
245 ID3D11VertexShader* VS;
246 ID3D11GeometryShader* GS;
247 UINT PSInstancesCount, VSInstancesCount, GSInstancesCount;
248 ID3D11ClassInstance *PSInstances[256], *VSInstances[256], *GSInstances[256]; // 256 is max according to PSSetShader documentation
249 D3D11_PRIMITIVE_TOPOLOGY PrimitiveTopology;
250 ID3D11Buffer* IndexBuffer, *VertexBuffer, *VSConstantBuffer;
251 UINT IndexBufferOffset, VertexBufferStride, VertexBufferOffset;
252 DXGI_FORMAT IndexBufferFormat;
253 ID3D11InputLayout* InputLayout;
254 };
255 BACKUP_DX11_STATE old = {};
256 old.ScissorRectsCount = old.ViewportsCount = D3D11_VIEWPORT_AND_SCISSORRECT_OBJECT_COUNT_PER_PIPELINE;
257 device->RSGetScissorRects(&old.ScissorRectsCount, old.ScissorRects);
258 device->RSGetViewports(&old.ViewportsCount, old.Viewports);
259 device->RSGetState(&old.RS);
260 device->OMGetBlendState(&old.BlendState, old.BlendFactor, &old.SampleMask);
261 device->OMGetDepthStencilState(&old.DepthStencilState, &old.StencilRef);
262 device->PSGetShaderResources(0, 1, &old.PSShaderResource);
263 device->PSGetSamplers(0, 1, &old.PSSampler);
264 old.PSInstancesCount = old.VSInstancesCount = old.GSInstancesCount = 256;
265 device->PSGetShader(&old.PS, old.PSInstances, &old.PSInstancesCount);
266 device->VSGetShader(&old.VS, old.VSInstances, &old.VSInstancesCount);
267 device->VSGetConstantBuffers(0, 1, &old.VSConstantBuffer);
268 device->GSGetShader(&old.GS, old.GSInstances, &old.GSInstancesCount);
269
270 device->IAGetPrimitiveTopology(&old.PrimitiveTopology);
271 device->IAGetIndexBuffer(&old.IndexBuffer, &old.IndexBufferFormat, &old.IndexBufferOffset);
272 device->IAGetVertexBuffers(0, 1, &old.VertexBuffer, &old.VertexBufferStride, &old.VertexBufferOffset);
273 device->IAGetInputLayout(&old.InputLayout);
274
275 // Setup desired DX state
276 ImGui_ImplDX11_SetupRenderState(draw_data, device);
277
278 // Setup render state structure (for callbacks and custom texture bindings)
279 ImGuiPlatformIO& platform_io = ImGui::GetPlatformIO();
280 ImGui_ImplDX11_RenderState render_state;
281 render_state.Device = bd->pd3dDevice;
282 render_state.DeviceContext = bd->pd3dDeviceContext;
283 render_state.SamplerDefault = bd->pFontSampler;
284 render_state.VertexConstantBuffer = bd->pVertexConstantBuffer;
285 platform_io.Renderer_RenderState = &render_state;
286
287 // Render command lists
288 // (Because we merged all buffers into a single one, we maintain our own offset into them)
289 int global_idx_offset = 0;
290 int global_vtx_offset = 0;
291 ImVec2 clip_off = draw_data->DisplayPos;
292 ImVec2 clip_scale = draw_data->FramebufferScale;
293 for (int n = 0; n < draw_data->CmdListsCount; n++)
294 {
295 const ImDrawList* draw_list = draw_data->CmdLists[n];
296 for (int cmd_i = 0; cmd_i < draw_list->CmdBuffer.Size; cmd_i++)
297 {
298 const ImDrawCmd* pcmd = &draw_list->CmdBuffer[cmd_i];
299 if (pcmd->UserCallback != nullptr)
300 {
301 // User callback, registered via ImDrawList::AddCallback()
302 // (ImDrawCallback_ResetRenderState is a special callback value used by the user to request the renderer to reset render state.)
303 if (pcmd->UserCallback == ImDrawCallback_ResetRenderState)
304 ImGui_ImplDX11_SetupRenderState(draw_data, device);
305 else
306 pcmd->UserCallback(draw_list, pcmd);
307 }
308 else
309 {
310 // Project scissor/clipping rectangles into framebuffer space
311 ImVec2 clip_min((pcmd->ClipRect.x - clip_off.x) * clip_scale.x, (pcmd->ClipRect.y - clip_off.y) * clip_scale.y);
312 ImVec2 clip_max((pcmd->ClipRect.z - clip_off.x) * clip_scale.x, (pcmd->ClipRect.w - clip_off.y) * clip_scale.y);
313 if (clip_max.x <= clip_min.x || clip_max.y <= clip_min.y)
314 continue;
315
316 // Apply scissor/clipping rectangle
317 const D3D11_RECT r = { (LONG)clip_min.x, (LONG)clip_min.y, (LONG)clip_max.x, (LONG)clip_max.y };
318 device->RSSetScissorRects(1, &r);
319
320 // Bind texture, Draw
321 ID3D11ShaderResourceView* texture_srv = (ID3D11ShaderResourceView*)pcmd->GetTexID();
322 device->PSSetShaderResources(0, 1, &texture_srv);
323 device->DrawIndexed(pcmd->ElemCount, pcmd->IdxOffset + global_idx_offset, pcmd->VtxOffset + global_vtx_offset);
324 }
325 }
326 global_idx_offset += draw_list->IdxBuffer.Size;
327 global_vtx_offset += draw_list->VtxBuffer.Size;
328 }
329 platform_io.Renderer_RenderState = nullptr;
330
331 // Restore modified DX state
332 device->RSSetScissorRects(old.ScissorRectsCount, old.ScissorRects);
333 device->RSSetViewports(old.ViewportsCount, old.Viewports);
334 device->RSSetState(old.RS); if (old.RS) old.RS->Release();
335 device->OMSetBlendState(old.BlendState, old.BlendFactor, old.SampleMask); if (old.BlendState) old.BlendState->Release();
336 device->OMSetDepthStencilState(old.DepthStencilState, old.StencilRef); if (old.DepthStencilState) old.DepthStencilState->Release();
337 device->PSSetShaderResources(0, 1, &old.PSShaderResource); if (old.PSShaderResource) old.PSShaderResource->Release();
338 device->PSSetSamplers(0, 1, &old.PSSampler); if (old.PSSampler) old.PSSampler->Release();
339 device->PSSetShader(old.PS, old.PSInstances, old.PSInstancesCount); if (old.PS) old.PS->Release();
340 for (UINT i = 0; i < old.PSInstancesCount; i++) if (old.PSInstances[i]) old.PSInstances[i]->Release();
341 device->VSSetShader(old.VS, old.VSInstances, old.VSInstancesCount); if (old.VS) old.VS->Release();
342 device->VSSetConstantBuffers(0, 1, &old.VSConstantBuffer); if (old.VSConstantBuffer) old.VSConstantBuffer->Release();
343 device->GSSetShader(old.GS, old.GSInstances, old.GSInstancesCount); if (old.GS) old.GS->Release();
344 for (UINT i = 0; i < old.VSInstancesCount; i++) if (old.VSInstances[i]) old.VSInstances[i]->Release();
345 device->IASetPrimitiveTopology(old.PrimitiveTopology);
346 device->IASetIndexBuffer(old.IndexBuffer, old.IndexBufferFormat, old.IndexBufferOffset); if (old.IndexBuffer) old.IndexBuffer->Release();
347 device->IASetVertexBuffers(0, 1, &old.VertexBuffer, &old.VertexBufferStride, &old.VertexBufferOffset); if (old.VertexBuffer) old.VertexBuffer->Release();
348 device->IASetInputLayout(old.InputLayout); if (old.InputLayout) old.InputLayout->Release();
349}

351static void ImGui_ImplDX11_DestroyTexture(ImTextureData* tex)
352{
353 ImGui_ImplDX11_Texture* backend_tex = (ImGui_ImplDX11_Texture*)tex->BackendUserData;
354 if (backend_tex == nullptr)
355 return;
356 IM_ASSERT(backend_tex->pTextureView == (ID3D11ShaderResourceView*)(intptr_t)tex->TexID);
357 backend_tex->pTextureView->Release();
358 backend_tex->pTexture->Release();
359 IM_DELETE(backend_tex);
360
361 // Clear identifiers and mark as destroyed (in order to allow e.g. calling InvalidateDeviceObjects while running)
362 tex->SetTexID(ImTextureID_Invalid);
363 tex->SetStatus(ImTextureStatus_Destroyed);
364 tex->BackendUserData = nullptr;
365}

367void ImGui_ImplDX11_UpdateTexture(ImTextureData* tex)
368{
369 ImGui_ImplDX11_Data* bd = ImGui_ImplDX11_GetBackendData();
370 if (tex->Status == ImTextureStatus_WantCreate)
371 {
372 // Create and upload new texture to graphics system
373 //IMGUI_DEBUG_LOG("UpdateTexture #%03d: WantCreate %dx%d\n", tex->UniqueID, tex->Width, tex->Height);
374 IM_ASSERT(tex->TexID == ImTextureID_Invalid && tex->BackendUserData == nullptr);
375 IM_ASSERT(tex->Format == ImTextureFormat_RGBA32);
376 unsigned int* pixels = (unsigned int*)tex->GetPixels();
377 ImGui_ImplDX11_Texture* backend_tex = IM_NEW(ImGui_ImplDX11_Texture)();
378
379 // Create texture
380 D3D11_TEXTURE2D_DESC desc;
381 ZeroMemory(&desc, sizeof(desc));
382 desc.Width = (UINT)tex->Width;
383 desc.Height = (UINT)tex->Height;
384 desc.MipLevels = 1;
385 desc.ArraySize = 1;
386 desc.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
387 desc.SampleDesc.Count = 1;
388 desc.Usage = D3D11_USAGE_DEFAULT;
389 desc.BindFlags = D3D11_BIND_SHADER_RESOURCE;
390 desc.CPUAccessFlags = 0;
391 D3D11_SUBRESOURCE_DATA subResource;
392 subResource.pSysMem = pixels;
393 subResource.SysMemPitch = desc.Width * 4;
394 subResource.SysMemSlicePitch = 0;
395 bd->pd3dDevice->CreateTexture2D(&desc, &subResource, &backend_tex->pTexture);
396 IM_ASSERT(backend_tex->pTexture != nullptr && "Backend failed to create texture!");
397
398 // Create texture view
399 D3D11_SHADER_RESOURCE_VIEW_DESC srvDesc;
400 ZeroMemory(&srvDesc, sizeof(srvDesc));
401 srvDesc.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
402 srvDesc.ViewDimension = D3D11_SRV_DIMENSION_TEXTURE2D;
403 srvDesc.Texture2D.MipLevels = desc.MipLevels;
404 srvDesc.Texture2D.MostDetailedMip = 0;
405 bd->pd3dDevice->CreateShaderResourceView(backend_tex->pTexture, &srvDesc, &backend_tex->pTextureView);
406 IM_ASSERT(backend_tex->pTextureView != nullptr && "Backend failed to create texture!");
407
408 // Store identifiers
409 tex->SetTexID((ImTextureID)(intptr_t)backend_tex->pTextureView);
410 tex->SetStatus(ImTextureStatus_OK);
411 tex->BackendUserData = backend_tex;
412 }
413 else if (tex->Status == ImTextureStatus_WantUpdates)
414 {
415 // Update selected blocks. We only ever write to textures regions which have never been used before!
416 // This backend choose to use tex->Updates[] but you can use tex->UpdateRect to upload a single region.
417 ImGui_ImplDX11_Texture* backend_tex = (ImGui_ImplDX11_Texture*)tex->BackendUserData;
418 IM_ASSERT(backend_tex->pTextureView == (ID3D11ShaderResourceView*)(intptr_t)tex->TexID);
419 for (ImTextureRect& r : tex->Updates)
420 {
421 D3D11_BOX box = { (UINT)r.x, (UINT)r.y, (UINT)0, (UINT)(r.x + r.w), (UINT)(r.y + r .h), (UINT)1 };
422 bd->pd3dDeviceContext->UpdateSubresource(backend_tex->pTexture, 0, &box, tex->GetPixelsAt(r.x, r.y), (UINT)tex->GetPitch(), 0);
423 }
424 tex->SetStatus(ImTextureStatus_OK);
425 }
426 if (tex->Status == ImTextureStatus_WantDestroy && tex->UnusedFrames > 0)

431{

433 if (!bd->pd3dDevice)
434 return false;

436
437 // By using D3DCompile() from <d3dcompiler.h> / d3dcompiler.lib, we introduce a dependency to a given version of d3dcompiler_XX.dll (see D3DCOMPILER_DLL_A)
438 // If you would like to use this DX11 sample code but remove this dependency you can:
439 // 1) compile once, save the compiled shader blobs into a file or source code and pass them to CreateVertexShader()/CreatePixelShader() [preferred solution]
440 // 2) use code to detect any version of the DLL and grab a pointer to D3DCompile from the DLL.
441 // See https://github.com/ocornut/imgui/pull/638 for sources and details.
442
443 // Create the vertex shader
444 {
445 static const char* vertexShader =
446 "cbuffer vertexBuffer : register(b0) \
447 {\
448 float4x4 ProjectionMatrix; \
449 };\
450 struct VS_INPUT\
451 {\
452 float2 pos : POSITION;\
453 float4 col : COLOR0;\
454 float2 uv : TEXCOORD0;\
455 };\
456 \
457 struct PS_INPUT\
458 {\
459 float4 pos : SV_POSITION;\
460 float4 col : COLOR0;\
461 float2 uv : TEXCOORD0;\
462 };\
463 \
464 PS_INPUT main(VS_INPUT input)\
465 {\
466 PS_INPUT output;\
467 output.pos = mul( ProjectionMatrix, float4(input.pos.xy, 0.f, 1.f));\
468 output.col = input.col;\
469 output.uv = input.uv;\
470 return output;\
471 }";
472
473 ID3DBlob* vertexShaderBlob;
474 if (FAILED(D3DCompile(vertexShader, strlen(vertexShader), nullptr, nullptr, nullptr, "main", "vs_4_0", 0, 0, &vertexShaderBlob, nullptr)))
475 return false; // NB: Pass ID3DBlob* pErrorBlob to D3DCompile() to get error showing in (const char*)pErrorBlob->GetBufferPointer(). Make sure to Release() the blob!
476 if (bd->pd3dDevice->CreateVertexShader(vertexShaderBlob->GetBufferPointer(), vertexShaderBlob->GetBufferSize(), nullptr, &bd->pVertexShader) != S_OK)

482 // Create the input layout

484 {

486 { "TEXCOORD", 0, DXGI_FORMAT_R32G32_FLOAT, 0, (UINT)offsetof(ImDrawVert, uv), D3D11_INPUT_PER_VERTEX_DATA, 0 },
487 { "COLOR", 0, DXGI_FORMAT_R8G8B8A8_UNORM, 0, (UINT)offsetof(ImDrawVert, col), D3D11_INPUT_PER_VERTEX_DATA, 0 },
488 };
489 if (bd->pd3dDevice->CreateInputLayout(local_layout, 3, vertexShaderBlob->GetBufferPointer(), vertexShaderBlob->GetBufferSize(), &bd->pInputLayout) != S_OK)

508 // Create the pixel shader
509 {
510 static const char* pixelShader =
511 "struct PS_INPUT\
512 {\
513 float4 pos : SV_POSITION;\
514 float4 col : COLOR0;\
515 float2 uv : TEXCOORD0;\
516 };\
517 sampler sampler0;\
518 Texture2D texture0;\
519 \
520 float4 main(PS_INPUT input) : SV_Target\
521 {\
522 float4 out_col = input.col * texture0.Sample(sampler0, input.uv); \
523 return out_col; \
524 }";
525
526 ID3DBlob* pixelShaderBlob;
527 if (FAILED(D3DCompile(pixelShader, strlen(pixelShader), nullptr, nullptr, nullptr, "main", "ps_4_0", 0, 0, &pixelShaderBlob, nullptr)))

530 {
531 pixelShaderBlob->Release();
532 return false;
533 }
534 pixelShaderBlob->Release();
535 }

537 // Create the blending setup
538 {
539 D3D11_BLEND_DESC desc;
540 ZeroMemory(&desc, sizeof(desc));
541 desc.AlphaToCoverageEnable = false;
542 desc.RenderTarget[0].BlendEnable = true;
543 desc.RenderTarget[0].SrcBlend = D3D11_BLEND_SRC_ALPHA;
544 desc.RenderTarget[0].DestBlend = D3D11_BLEND_INV_SRC_ALPHA;
545 desc.RenderTarget[0].BlendOp = D3D11_BLEND_OP_ADD;
546 desc.RenderTarget[0].SrcBlendAlpha = D3D11_BLEND_ONE;
547 desc.RenderTarget[0].DestBlendAlpha = D3D11_BLEND_INV_SRC_ALPHA;
548 desc.RenderTarget[0].BlendOpAlpha = D3D11_BLEND_OP_ADD;
549 desc.RenderTarget[0].RenderTargetWriteMask = D3D11_COLOR_WRITE_ENABLE_ALL;
550 bd->pd3dDevice->CreateBlendState(&desc, &bd->pBlendState);

553 // Create the rasterizer state
554 {

703// Multi-Viewports: configure templates used when creating swapchains for secondary viewports. Will try them in order.

713static void ImGui_ImplDX11_CreateWindow(ImGuiViewport* viewport)
714{
715 ImGui_ImplDX11_Data* bd = ImGui_ImplDX11_GetBackendData();
716 ImGui_ImplDX11_ViewportData* vd = IM_NEW(ImGui_ImplDX11_ViewportData)();
717 viewport->RendererUserData = vd;
718
719 // PlatformHandleRaw should always be a HWND, whereas PlatformHandle might be a higher-level handle (e.g. GLFWWindow*, SDL's WindowID).
720 // Some backends will leave PlatformHandleRaw == 0, in which case we assume PlatformHandle will contain the HWND.
721 HWND hwnd = viewport->PlatformHandleRaw ? (HWND)viewport->PlatformHandleRaw : (HWND)viewport->PlatformHandle;
722 IM_ASSERT(hwnd != 0);
723 IM_ASSERT(vd->SwapChain == nullptr && vd->RTView == nullptr);
724
725 // Create swap chain
726 HRESULT hr = DXGI_ERROR_UNSUPPORTED;
727 for (const DXGI_SWAP_CHAIN_DESC& sd_template : bd->SwapChainDescsForViewports)
728 {
729 IM_ASSERT(sd_template.BufferDesc.Width == 0 && sd_template.BufferDesc.Height == 0 && sd_template.OutputWindow == nullptr);
730 DXGI_SWAP_CHAIN_DESC sd = sd_template;
731 sd.BufferDesc.Width = (UINT)viewport->Size.x;
732 sd.BufferDesc.Height = (UINT)viewport->Size.y;
733 sd.OutputWindow = hwnd;
734 hr = bd->pFactory->CreateSwapChain(bd->pd3dDevice, &sd, &vd->SwapChain);
735 if (SUCCEEDED(hr))
736 break;
737 }
738 IM_ASSERT(SUCCEEDED(hr));
739
740 // Create the render target
741 if (vd->SwapChain != nullptr)
742 {
743 ID3D11Texture2D* pBackBuffer;
744 vd->SwapChain->GetBuffer(0, IID_PPV_ARGS(&pBackBuffer));
745 bd->pd3dDevice->CreateRenderTargetView(pBackBuffer, nullptr, &vd->RTView);
746 pBackBuffer->Release();
747 }
748}
