146int mtgb::TextCache::GetOrCreateTextLayoutHandle(const std::wstring& text, int size, float width, float height)
147{
148 // text + size の範囲を取得
149 auto& text_index = textLayoutData_->get<text_layout_order>();
150 auto range = text_index.equal_range(std::make_tuple(text, size));
151 for (auto it = range.first; it != range.second; ++it)
152 {
153 // width / height も一致するものを探す
154 TextLayoutData* data = *it;
155 if (data->width == width && data->height == height)
156 {
157 return data->handle; // 完全一致があれば再利用
158 }
159 }
160
161 // 新規作成する
162 FontFormatData* formatData = GetOrCreateTextFormat(size);
163
164 IDWriteTextLayout* pTextLayout = nullptr;
165 Game::System<DirectWrite>().CreateTextLayout(text, width, height, size, formatData->format, &pTextLayout);
166
167 int handle = ++nextHandle_;
168 TextLayoutData* layoutData = new TextLayoutData{ text,size, width,height,pTextLayout,handle };
169 textLayoutData_->insert(layoutData);
170 return handle;
171}
