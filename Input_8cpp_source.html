375BOOL CALLBACK EnumJoysticksCallback(const LPCDIDEVICEINSTANCE lpddi, LPVOID pvRef)
376{
377 auto& input = Game::System<Input>();
378
379 // 割り当て予約がなかったらデバイスを作成しない
380 if (input.IsNotSubscribed())
381 {
382 return DIENUM_STOP;
383 }
384 DeviceType devType = Input::GetDeviceType(*lpddi);
385
386 int reservationIndex = input.FindReservationIndexForDevice(devType);
387
388 if(reservationIndex < 0)
389 {
390 // デバイスを要求している予約はない
391 return DIENUM_CONTINUE;
392 }
393 LPDIRECTINPUT8 pDirectInput = reinterpret_cast<LPDIRECTINPUT8>(pvRef);
394 ComPtr<IDirectInputDevice8> pDevice = nullptr;
395
396 // 割り当て予約があり、未割当なのでデバイス作成
397 HRESULT hResult = pDirectInput->CreateDevice(lpddi->guidInstance, pDevice.GetAddressOf(), nullptr);
398 massert(SUCCEEDED(hResult)
399 && "ジョイスティックのデバイスの作成に失敗 @EnumJoysticksCallback");
400
401 if (Game::System<Input>().RegisterJoystickGuid(lpddi->guidInstance) == false)
402 {
403 //既に割り当て済みの為、他のデバイスの列挙に移す
404 pDevice.Reset();
405 return DIENUM_CONTINUE;
406 }
407
408 input.AssignJoystickToReservation(pDevice, static_cast<size_t>(reservationIndex), lpddi->guidInstance);
409 LOGIMGUI_CAT("Input", "Assigned reservationIndex=%d", reservationIndex);
410
411 // 予約がまだ残っているなら続行
412 return input.IsNotSubscribed() ? DIENUM_STOP : DIENUM_CONTINUE;
413}

577{
578 DeviceType deviceType = DeviceType::Unknown;
579
580 //REF:https://learn.microsoft.com/ja-jp/previous-versions/windows/desktop/ee416610(v=vs.85)?devlangs=cpp&f1url=%3FappId%3DDev17IDEF1%26l%3DJA-JP%26k%3Dk(DINPUT%2FDIDEVICEINSTANCE)%3Bk(DIDEVICEINSTANCE)%3Bk(DevLang-C%2B%2B)%3Bk(TargetOS-Windows)%26rd%3Dtrue
581 // 下位ビットでデバイスの大まかなタイプを判別
582 // 上位ビットでデバイスのサブタイプも判別できるよ
583 DWORD major = _inst.dwDevType & 0xFF;
584 switch (major)
585 {
586 case DI8DEVTYPE_FLIGHT:
587 deviceType = DeviceType::FlightStick;
588 break;
589 case DI8DEVTYPE_GAMEPAD:
590 case DI8DEVTYPE_JOYSTICK:
591 case DI8DEVTYPE_1STPERSON:
592 deviceType = DeviceType::GamePad;
593 break;
594 default:
595 deviceType = DeviceType::Unknown;
596 break;
597 }
598
599 return deviceType;

683void mtgb::Input::SetProperty(ComPtr<IDirectInputDevice8> _pJoystickDevice, InputConfig _inputConfig)
684{
685 HRESULT hResult{};
686
687#pragma region 軸モード設定
688
689 DIPROPDWORD diprop;
690 diprop.diph.dwSize = sizeof(diprop);
691 diprop.diph.dwHeaderSize = sizeof(diprop.diph);
692 diprop.diph.dwHow = DIPH_DEVICE;
693
694 //https://learn.microsoft.com/ja-jp/previous-versions/windows/desktop/ee416636(v=vs.85)
695 //dwHowがDIPH_DEVICEの場合は0にしないといけない
696 diprop.diph.dwObj = 0;
697
698 //REL:前回のデバイスとの相対値を使用する
699 //ABS:デバイス上の絶対値を使用する
700 diprop.dwData = DIPROPAXISMODE_ABS;
