170void mtgb::ImGuiEditorCamera::DoDolly()
171{
172 Vector3 mouseMove = InputUtil::GetMouseMove();
173 if (mouseMove.Size() != 0)
174 {
175 // カメラの右、上ベクトル
176 Vector3 right = pCameraTransform_->Right();
177 Vector3 up = pCameraTransform_->Up();
178
179 // 移動量を合成
180 Vector3 move = right * -mouseMove.x + up * mouseMove.y;
181
182 pCameraTransform_->position += move * moveSpeed_ * Time::DeltaTimeF();
183 }

186void mtgb::ImGuiEditorCamera::DoPan()
187{
188 Vector3 mouseMove = InputUtil::GetMouseMove();
189 if (mouseMove.Size() != 0)
190 {
191 // マウス移動量を角度に変換
192 azimuthalAngleRad_ -= mouseMove.x * rotateSensitivity_ * Time::DeltaTimeF(); // 水平角度
193
194 polarAngleRad_ += mouseMove.y * rotateSensitivity_ * Time::DeltaTimeF(); // 鉛直角度
195
196 // 鉛直角度を制限
197 polarAngleRad_ = std::clamp(polarAngleRad_, DirectX::XMConvertToRadians(0.1f), DirectX::XMConvertToRadians(179.0f));
198
199 MoveCameraSphericalOnTheSpot();
200 }

216void mtgb::ImGuiEditorCamera::MoveCameraSphericalOnTheSpot()
217{
218 // ref:https://ja.wikipedia.org/wiki/%E7%90%83%E9%9D%A2%E5%BA%A7%E6%A8%99%E7%B3%BB
219
220 // θ (polar angle) : 鉛直方向
221 float theta = polarAngleRad_;
222
223 // φ (azimuthal angle): 水平方向
224 float phi = azimuthalAngleRad_;
225
226 // 回転中心からのオフセット
227 Vector3 offset;
228
229 // 変換
230 offset.x = sinf(theta) * cos(phi);
231 offset.y = cos(theta);
232 offset.z = sin(theta) * sin(phi);
233
234 // その場回転の時はoffsetの方向を向く
235 pCameraTransform_->rotate = Quaternion::LookRotation(offset, Vector3::Up());
236}

238void mtgb::ImGuiEditorCamera::SelectTransform()
239{
240 Vector3 origin, end, vec;
241 Matrix4x4 proj, view;
242 Game::System<CameraSystem>().GetProjMatrix(&proj);
243 Game::System<CameraSystem>().GetViewMatrix(&view);
244
245 ImGuiWindow* window = ImGui::FindWindowByName(windowName_.c_str());
246
247 if (window == nullptr) return;
248
249 ImRect workRect = window->WorkRect;
250 ImVec2 workPos = workRect.Min;
251 ImGuiUtil::GetMouseRay(origin, end, proj, view, Game::System<ImGuiRenderer>().GetViewport(), {workPos.x,workPos.y});
252
253 vec = end - origin;
254
255 // vec.Normalize()の結果を別変数に保存して、元の長さを保持
256 Vector3 direction = vec.Normalize(); // これで正規化されたベクトルが返される
257
258 const CameraSystem& camera = Game::System<CameraSystem>();
259 float distance = camera.GetFar() - camera.GetNear(); // 元の長さを計算
260
261 EntityId entityId = Game::System<ColliderCP>().RayCastHitAll(origin, direction, distance);
262 if (entityId != INVALID_ENTITY)
263 {
264 // EntityがTransformコンポーネントを持っていない可能性があるのでTryGet
265 Game::System<TransformCP>().TryGet(pTargetTransform_, entityId);
266
267 mtgb::GameObjectSelectedEvent event{ .entityId = entityId };
268 Game::System<EventManager>().GetEvent<mtgb::GameObjectSelectedEvent>().Invoke(event);
269 LOGIMGUI("EditorCamera:Selected");
270 }
271 else
272 {
273 mtgb::GameObjectDeselectedEvent event;
274 if (pTargetTransform_ != nullptr)
275 {
276 event = { .entityId = pTargetTransform_->GetEntityId()};
277 pTargetTransform_ = nullptr;
278 }
279 else
280 {
281 event = { .entityId = INVALID_ENTITY };
282 }
283
284 Game::System<EventManager>().GetEvent<mtgb::GameObjectDeselectedEvent>().Invoke(event);
285 LOGIMGUI("EditorCamera:No Select");
286
287 }
