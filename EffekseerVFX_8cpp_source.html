127 void EffectManager::Update()
128 {
129 double _deltaTime = Time::DeltaTime();
130 for (auto iter = effectInstances_.begin(); iter != effectInstances_.end();)
131 {
132 auto& instance = *iter->second;
133 auto& handle = iter->second->handle;
134 auto& tranform = iter->second->effectTransform;
135
136 // 経過時間が0の場合、再生を始める
137 if (instance.elapsedTime == 0)
138 {
139 handle = managerRef_->Play(instance.GetEffectData()->GetEffectRef(), 0, 0, 0);
140 }
141
142 int32_t maxFrame = GetMaxFrame(instance.GetEffectData()->GetEffectRef());
143 // 再生時間
144 double duration = maxFrame / fps_;
145 if (instance.elapsedTime > duration)
146 {
147 // エフェクトの再生を終了
148 managerRef_->StopEffect(handle);
149
150 // ループを行うか
151 if (tranform->isLoop)
152 {
153 // 経過時間を0にして、また再生されるように設定
154 instance.elapsedTime = 0;
155 }
156 else
157 {
158 // ループを行わない場合、不要なので削除
159 iter = effectInstances_.erase(iter);
160 }
161 }
162 // エフェクトの更新を行う
163 else
164 {
165 // フレームを設定
166 rendererRef_->SetTime(static_cast<float>(instance.elapsedTime));
167 // エフェクトのインスタンスに対して行列、更新速度を設定
168 managerRef_->SetMatrix(handle, CnvMat43(tranform->worldMat));
169 managerRef_->SetSpeed(handle, tranform->speed);
170 // 経過時間を進める
171 instance.elapsedTime += _deltaTime;
172 ++iter;
173 }
174 }
175
176 managerRef_->Update(static_cast<float>(_deltaTime * fps_));
177 }
