519inline int32_t GetTextureMemorySize(TextureFormatType format, Vec3I size)
520{
521 switch (format)
522 {
523 case TextureFormatType::R8G8B8A8_UNORM:
524 return size.X * size.Y * size.Z * 4;
525 case TextureFormatType::B8G8R8A8_UNORM:
526 return size.X * size.Y * size.Z * 4;
527 case TextureFormatType::R8_UNORM:
528 return size.X * size.Y * size.Z * 1;
529 case TextureFormatType::R16G16_FLOAT:
530 return size.X * size.Y * size.Z * 4;
531 case TextureFormatType::R16G16B16A16_FLOAT:
532 return size.X * size.Y * size.Z * 8;
533 case TextureFormatType::R32G32B32A32_FLOAT:
534 return size.X * size.Y * size.Z * 16;
535 case TextureFormatType::R8G8B8A8_UNORM_SRGB:
536 return size.X * size.Y * size.Z * 4;
537 case TextureFormatType::B8G8R8A8_UNORM_SRGB:
538 return size.X * size.Y * size.Z * 4;
539 case TextureFormatType::D32:
540 return size.X * size.Y * size.Z * 4;
541 case TextureFormatType::D24S8:
542 return size.X * size.Y * size.Z * 4;
543 case TextureFormatType::D32S8:
544 return size.X * size.Y * size.Z * 5;
545 default:
546 auto str = to_string(format);
547 Log(LogType::Error, str + " : GetTextureMemorySize is not supported");
548 return 0;
549 }
