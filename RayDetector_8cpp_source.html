93 bool RayDetector::IsTargetInRayAngle(const Vector3& targetPos, const Vector3& rayOrigin,
94 const Vector3& rayDirection, float maxAngleDegrees) const
95 {
96 // レイの起点からターゲットへのベクトル
97 Vector3 toTarget = Vector3::Normalize(targetPos - rayOrigin);
98
99 // レイの方向ベクトルを正規化
100 Vector3 normalizedRayDir = Vector3::Normalize(rayDirection);
101
102 // 内積を使って角度を計算
103 float dotProduct = DirectX::XMVectorGetX(DirectX::XMVector3Dot(normalizedRayDir, toTarget));
104
105 // 内積から角度を計算（ラジアン）
106 float angleRadians = std::acosf(std::clamp(dotProduct, -1.0f, 1.0f));
107
108 // 度に変換
109 float angleDegrees = DirectX::XMConvertToDegrees(angleRadians);
110
111 // 指定角度以内かチェック
112 return angleDegrees <= maxAngleDegrees;
113 }
