34LRESULT WindowResource::WndProc(HWND hWnd, UINT msg, WPARAM wParam, LPARAM lParam)
35{
36 WindowResource* pThis;
37 if (msg == WM_NCCREATE)
38 {
39 //作成時のパラメータからthisを取得、キャスト
40 pThis = static_cast<WindowResource*>(reinterpret_cast<LPCREATESTRUCT>(lParam)->lpCreateParams);
41 //thisのUSERDATAにthisを紐付ける
42 SetWindowLongPtr(hWnd, GWLP_USERDATA, reinterpret_cast<LONG_PTR>(pThis));
43 pThis->hWnd_ = hWnd;
44 }
45 else
46 {
47 //hWndにthisを紐づけておいたので取得
48 pThis = reinterpret_cast<WindowResource*>(GetWindowLongPtr(hWnd, GWLP_USERDATA));
49 if (pThis)
50 {
51 pThis->hWnd_ = hWnd;
52 }
53 }
54 if (pThis)
55 {
56 return pThis->HandleWindowMessage(hWnd, msg, wParam, lParam);
57 }
58 return DefWindowProc(hWnd, msg, wParam, lParam);

60}

62LRESULT WindowResource::HandleWindowMessage(HWND hWnd, UINT msg, WPARAM wParam, LPARAM lParam)
63{
64 if (ImGui_ImplWin32_WndProcHandler(hWnd, msg, wParam, lParam))
65 {
66 return true;
67 }
68
69
70 switch (msg)
71 {
72 case WM_CLOSE:
73 Game::Exit();
74 return 0;
75 case WM_DESTROY: // ウィンドウを閉じた
76 return S_OK;
77 case WM_MOUSEMOVE: // マウスが動いた
78 Game::System<Input>().UpdateMousePositionData(LOWORD(lParam), HIWORD(lParam));
79 return S_OK;
80 case WM_SIZE: // ウィンドウサイズが変わった
81 {
82 if (this && wParam != SIZE_MINIMIZED)
83 {
84 if (!isInitialized_)
85 {
86 // まだ初期化されていないならスキップする
87 return S_OK;
88 }
89
90 UINT width = LOWORD(lParam);
91 UINT height = HIWORD(lParam);
92 }
93 return S_OK;
94 }
95
96 default: // それ以外のメッセージは譲渡
97 break;
98 }
99
100
101 // NOTE: これが抜けているとウィンドウ表示されないし、エラーは出ないしで苦労する(した)
102 return DefWindowProc(hWnd, msg, wParam, lParam);
103}
