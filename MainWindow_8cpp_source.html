149static LRESULT CALLBACK WndProc(HWND hWnd, UINT message, WPARAM wParam, LPARAM lParam)
150{
151 if (ImGui_ImplWin32_WndProcHandler(hWnd, message, wParam, lParam))
152 {
153 return true;
154 }
155 switch (message)
156 {
157 case WM_DESTROY: // ウィンドウを閉じた
158 Game::Exit();
159 return S_OK;
160 case WM_MOUSEMOVE: // マウスが動いた
161 Game::System<Input>().UpdateMousePositionData(LOWORD(lParam), HIWORD(lParam));
162 return S_OK;
163 case WM_SIZE: // ウィンドウサイズが変わった
164
165 return S_OK;
166 case WM_NCCALCSIZE:
167 // wParamについて：TRUEならNCCALCSIZE_PARAMS / FALSEならRect*
168 if (wParam) return 0;
169 break;
170 case WM_NCHITTEST:
171 {
172 LRESULT hitResult{ DefWindowProc(hWnd, message, wParam, lParam) };
173 switch (hitResult)
174 {
175 case HTLEFT:
176 case HTRIGHT:
177 case HTTOP:
178 case HTBOTTOM:
179 case HTTOPLEFT:
180 case HTTOPRIGHT:
181 case HTBOTTOMLEFT:
182 case HTBOTTOMRIGHT:
183 return HTCLIENT; // サイズは変えさせない
184 default:
185 return hitResult;
186 }
187 }
188 default: // それ以外のメッセージは譲渡
189 break;
190 }
191
192
193 // NOTE: これが抜けているとウィンドウ表示されないし、エラーは出ないしで苦労する(した)
194 return DefWindowProc(hWnd, message, wParam, lParam);
195}
