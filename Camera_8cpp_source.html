161void mtgb::Camera::MoveCameraSpherical(float _distance)
162{
163 if (pTargetTransform_ == nullptr)
164 return;
165
166 // 注視点を計算
167 float lerpedY = std::lerp(baseY, distY,lookAtPosLerpProgress);
168 Vector3 lookAtTarget = Vector3(pTargetTransform_->position.x, lerpedY,pTargetTransform_->position.z) + lookAtPositionOffset_;
169
170 // 球面座標で新しいカメラ位置を計算
171 float theta = polarAngleRad_;
172 float phi = azimuthalAngleRad_;
173
174 Vector3 offset;
175 offset.x = _distance * sinf(theta) * cosf(phi);
176 offset.y = -_distance * cosf(theta);
177 offset.z = -_distance * sinf(theta) * sinf(phi);
178
179 Vector3 newCameraPos = lookAtTarget + offset;
180
181 pCameraTransform_->position = newCameraPos;
182
183 // カメラの回転を設定
184 Vector3 lookDir = lookAtTarget - pCameraTransform_->position;
185 pCameraTransform_->rotate = Quaternion::LookRotation(lookDir.Normalize(), Vector3::Up());
186}
