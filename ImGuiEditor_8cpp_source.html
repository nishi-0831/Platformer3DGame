110void mtgb::ImGuiEditor::LoadMapData()
111{
112 TCHAR fileName[255] = "";
113 OPENFILENAME ifn = { 0 };
114
115 ifn.lStructSize = sizeof(ifn);
116 ifn.hwndOwner = WinCtxRes::GetHWND(WindowContext::First);
117 ifn.lpstrFilter = "JSONファイル(*.json)\0*.json";
118 ifn.lpstrFile = fileName;
119 ifn.nMaxFile = 255;
120
121 if (GetOpenFileName(&ifn) == false)
122 return;
123
124 std::ifstream inputFile(fileName);
125 if (inputFile.fail())
126 {
127 assert(false);
128 }
129 nlohmann::json json;
130 try
131 {
132 inputFile >> json;
133 }
134 catch (const nlohmann::json::parse_error& e)
135 {
136 assert(false,e.what());
137 }
138 GameObjectGenerator::GenerateFromJson(json);
139
140 // 読み込み時間で値が大きくなったデルタタイムを安定させるために2フレーム待機させる
141 // TODO: マジックナンバーを修正
