28 // 現在のカメラ位置から回転中心を計算
29 Vector3 center = Vector3::Zero();
30 // ターゲットを中心に回転
31 if (pTargetTransform_)
32 {
33 Vector3 toTarget = pTargetTransform_->position - pCameraTransform_->position;
34 center = pTargetTransform_->position + (Vector3::Normalize(-toTarget) * _distance) + lookAtPositionOffset_;
35 }
36 // ターゲットがいない場合は正面を向いて回転
37 else
38 {
39 center = pCameraTransform_->position + (pCameraTransform_->Forward() * _distance) + lookAtPositionOffset_;
40 }
41
42 // θ (polar angle) : 鉛直方向
43 float theta = polarAngleRad_;
44
45 // φ (azimuthal angle): 水平方向
46 float phi = azimuthalAngleRad_;
47
48 // 回転中心からのオフセット
49 Vector3 offset;
50
51 // 回転中心の方向を向く
52
53 // 変換
54 offset.x = _distance * sinf(theta) * cos(phi);
55 offset.y = -_distance * cos(theta);
56 offset.z = -_distance * sin(theta) * sin(phi);
57
58 // 位置を反映
59 pCameraTransform_->position = center + offset;
60
