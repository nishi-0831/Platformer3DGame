190const float mtgb::InputUtil::GetAxis(Axis _axis, StickType _stickType,WindowContext _context)
191{
192 const InputData& input = GetInput(_context);
193 float value = 0.0f;
194 switch (_axis)
195 {
196 case Axis::X:
197 if (_stickType == StickType::LEFT)
198 {
199 value = static_cast<float>(input.joyStateCurrent_.lX);
200 }
201 else
202 {
203 if (input.controllerType_ == ControllerType::DualShock)
204 {
205 value = static_cast<float>(input.joyStateCurrent_.lZ);
206 }
207 else if (input.controllerType_ == ControllerType::Xbox)
208 {
209 value = static_cast<float>(input.joyStateCurrent_.lRx);
210 }
211 else
212 {
213 value = static_cast<float>(input.joyStateCurrent_.lZ);
214 }
215 }
216 value /= input.config_.xRange;
217 break;
218
219 case Axis::Y:
220 if (_stickType == StickType::LEFT)
221 {
222 value = static_cast<float>(input.joyStateCurrent_.lY);
223 }
224 else
225 {
226 if (input.controllerType_ == ControllerType::DualShock)
227 {
228 value = static_cast<float>(input.joyStateCurrent_.lRz);
229 }
230 else if (input.controllerType_ == ControllerType::Xbox)
231 {
232 value = -static_cast<float>(input.joyStateCurrent_.lRy);
233 }
234 else
235 {
236 value = static_cast<float>(input.joyStateCurrent_.lRz);
237 }
238 }
239 value /= input.config_.yRange;
240 break;
241 default: return 0.0f;
242 }
243 return input.config_.ApplyDeadZone(value);
