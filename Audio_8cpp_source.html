47void mtgb::Audio::Update()
48{
49 if (pOneShotQueue_.size() <= 0)
50 {
51 return; // キューが空なら回帰
52 }
53
54 // REF: CPUのSleepプロセス
55
56 // 再生終了をデキューする
57 auto itr = pOneShotQueue_.begin();
58 while (true)
59 {
60 (*itr)->timeLeft -= Time::DeltaTimeF();
61
62 float diff{ (*itr)->timeLeft }; // 差分
63 if (diff > 0) // 差分が0より大きい = 再生中
64 {
65 break; // 再生終了していないため離脱
66 }
67 else // 差分が0以下 = 再生終了
68 {
69 (*itr)->Release();
70 //delete (*itr); // 解放する
71 itr = pOneShotQueue_.erase(pOneShotQueue_.begin()); // 消す
72 if (itr == pOneShotQueue_.end())
73 {
74 break; // 後続がない = キューが空なら離脱
75 }
76 (*itr)->timeLeft += diff; // 差分を後続に反映
77 continue; // 後続も終了している可能性があるため継続
78 }
79 }

257void mtgb::Audio::EnqueueOneShot(AudioOneShot* _pOneShot)
258{
259 float lefter{ _pOneShot->timeLeft }; // 減算用
260 float righter{ 0 }; // 加算用
261 // 適切な挿入ポイントを見つける
262 for (auto itr = pOneShotQueue_.begin(); itr != pOneShotQueue_.end(); itr++)
263 {
264 if (lefter <= righter + (*itr)->timeLeft)
265 {
266 _pOneShot->timeLeft = lefter - righter;
267 itr = pOneShotQueue_.insert(itr, _pOneShot);
268 itr++;
269 (*itr)->timeLeft -= _pOneShot->timeLeft;
270 return;
271 }
272 righter += (*itr)->timeLeft;
273 }
274 // 見つからなかったら末端に追加
275 pOneShotQueue_.push_back(_pOneShot);
276}
