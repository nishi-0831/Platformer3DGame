40static inline float Distance(const ImVec2& a, const ImVec2& b)
41{
42 return sqrtf((a.x - b.x) * (a.x - b.x) + (a.y - b.y) * (a.y - b.y));
43}

50static ImVec2 GetInputSlotPos(Delegate& delegate, const Node& node, SlotIndex slotIndex, float factor)
51{
52 ImVec2 Size = node.mRect.GetSize() * factor;
53 size_t InputsCount = delegate.GetTemplate(node.mTemplateIndex).mInputCount;
54 return ImVec2(node.mRect.Min.x * factor,
55 node.mRect.Min.y * factor + Size.y * ((float)slotIndex + 1) / ((float)InputsCount + 1) + 8.f);
56}

58static ImVec2 GetOutputSlotPos(Delegate& delegate, const Node& node, SlotIndex slotIndex, float factor)
59{
60 ImVec2 Size = node.mRect.GetSize() * factor;
61 size_t OutputsCount = delegate.GetTemplate(node.mTemplateIndex).mOutputCount;
62 return ImVec2(node.mRect.Min.x * factor + Size.x,
63 node.mRect.Min.y * factor + Size.y * ((float)slotIndex + 1) / ((float)OutputsCount + 1) + 8.f);
64}

87static void HandleZoomScroll(ImRect regionRect, ViewState& viewState, const Options& options)
88{
89 ImGuiIO& io = ImGui::GetIO();
90
91 if (regionRect.Contains(io.MousePos))
92 {
93 if (io.MouseWheel < -FLT_EPSILON)
94 {
95 viewState.mFactorTarget *= 1.f - options.mZoomRatio;
96 }
97
98 if (io.MouseWheel > FLT_EPSILON)
99 {
100 viewState.mFactorTarget *= 1.0f + options.mZoomRatio;
101 }
102 }
103
104 ImVec2 mouseWPosPre = (io.MousePos - ImGui::GetCursorScreenPos()) / viewState.mFactor;
105 viewState.mFactorTarget = ImClamp(viewState.mFactorTarget, options.mMinZoom, options.mMaxZoom);
106 viewState.mFactor = ImLerp(viewState.mFactor, viewState.mFactorTarget, options.mZoomLerpFactor);
107 ImVec2 mouseWPosPost = (io.MousePos - ImGui::GetCursorScreenPos()) / viewState.mFactor;
108 if (ImGui::IsMousePosValid())
109 {
110 viewState.mPosition += mouseWPosPost - mouseWPosPre;
111 }

119static void FitNodes(Delegate& delegate, ViewState& viewState, const ImVec2 viewSize, bool selectedNodesOnly)
120{
121 const size_t nodeCount = delegate.GetNodeCount();
122
123 if (!nodeCount)
124 {
125 return;
126 }
127
128 bool validNode = false;
129 ImVec2 min(FLT_MAX, FLT_MAX);
130 ImVec2 max(-FLT_MAX, -FLT_MAX);
131 for (NodeIndex nodeIndex = 0; nodeIndex < nodeCount; nodeIndex++)
132 {
133 const Node& node = delegate.GetNode(nodeIndex);
134
135 if (selectedNodesOnly && !node.mSelected)
136 {
137 continue;
138 }
139
140 min = ImMin(min, node.mRect.Min);
141 min = ImMin(min, node.mRect.Max);
142 max = ImMax(max, node.mRect.Min);
143 max = ImMax(max, node.mRect.Max);
144 validNode = true;
145 }
146
147 if (!validNode)
148 {
149 return;
150 }
151
152 min -= viewSize * 0.05f;
153 max += viewSize * 0.05f;
154 ImVec2 nodesSize = max - min;
155 ImVec2 nodeCenter = (max + min) * 0.5f;
156
157 float ratioY = viewSize.y / nodesSize.y;
158 float ratioX = viewSize.x / nodesSize.x;
159
160 viewState.mFactor = viewState.mFactorTarget = ImMin(ImMin(ratioY, ratioX), 1.f);
161 viewState.mPosition = ImVec2(-nodeCenter.x, -nodeCenter.y) + (viewSize * 0.5f) / viewState.mFactorTarget;
162}

350static bool HandleConnections(ImDrawList* drawList,
351 NodeIndex nodeIndex,
352 const ImVec2 offset,
353 const float factor,
354 Delegate& delegate,
355 const Options& options,
356 bool bDrawOnly,
357 SlotIndex& inputSlotOver,
358 SlotIndex& outputSlotOver,
359 const bool inMinimap)
360{
361 static NodeIndex editingNodeIndex;
362 static SlotIndex editingSlotIndex;
363
364 ImGuiIO& io = ImGui::GetIO();
365 const auto node = delegate.GetNode(nodeIndex);
366 const auto nodeTemplate = delegate.GetTemplate(node.mTemplateIndex);
367 const auto linkCount = delegate.GetLinkCount();
368
369 size_t InputsCount = nodeTemplate.mInputCount;
370 size_t OutputsCount = nodeTemplate.mOutputCount;
371 inputSlotOver = -1;
372 outputSlotOver = -1;
373
374 // draw/use inputs/outputs
375 bool hoverSlot = false;
376 for (int i = 0; i < 2; i++)
377 {
378 float closestDistance = FLT_MAX;
379 SlotIndex closestConn = -1;
380 ImVec2 closestTextPos;
381 ImVec2 closestPos;
382 const size_t slotCount[2] = {InputsCount, OutputsCount};
383
384 for (SlotIndex slotIndex = 0; slotIndex < slotCount[i]; slotIndex++)
385 {
386 const char** con = i ? nodeTemplate.mOutputNames : nodeTemplate.mInputNames;
387 const char* conText = (con && con[slotIndex]) ? con[slotIndex] : "";
388
389 ImVec2 p =
390 offset + (i ? GetOutputSlotPos(delegate, node, slotIndex, factor) : GetInputSlotPos(delegate, node, slotIndex, factor));
391 float distance = Distance(p, io.MousePos);
392 bool overCon = (nodeOperation == NO_None || nodeOperation == NO_EditingLink) &&
393 (distance < options.mNodeSlotRadius * 2.f) && (distance < closestDistance);
394
395
396 ImVec2 textSize;
397 textSize = ImGui::CalcTextSize(conText);
398 ImVec2 textPos =
399 p + ImVec2(-options.mNodeSlotRadius * (i ? -1.f : 1.f) * (overCon ? 3.f : 2.f) - (i ? 0 : textSize.x),
400 -textSize.y / 2);
401
402 ImRect nodeRect = GetNodeRect(node, factor);
403 if (!inMinimap && (overCon || (nodeRect.Contains(io.MousePos - offset) && closestConn == -1 &&
404 (editingInput == (i != 0)) && nodeOperation == NO_EditingLink)))
405 {
406 closestDistance = distance;
407 closestConn = slotIndex;
408 closestTextPos = textPos;
409 closestPos = p;
410
411 if (i)
412 {
413 outputSlotOver = slotIndex;
414 }
415 else
416 {
417 inputSlotOver = slotIndex;
418 }
419 }
420 else
421 {
422 const ImU32* slotColorSource = i ? nodeTemplate.mOutputColors : nodeTemplate.mInputColors;
423 const ImU32 slotColor = slotColorSource ? slotColorSource[slotIndex] : options.mDefaultSlotColor;
424 drawList->AddCircleFilled(p, options.mNodeSlotRadius, IM_COL32(0, 0, 0, 200));
425 drawList->AddCircleFilled(p, options.mNodeSlotRadius * 0.75f, slotColor);
426 if (!options.mDrawIONameOnHover)
427 {
428 drawList->AddText(io.FontDefault, 14, textPos + ImVec2(2, 2), IM_COL32(0, 0, 0, 255), conText);
429 drawList->AddText(io.FontDefault, 14, textPos, IM_COL32(150, 150, 150, 255), conText);
430 }
431 }
432 }
433
434 if (closestConn != -1)
435 {
436 const char** con = i ? nodeTemplate.mOutputNames : nodeTemplate.mInputNames;
437 const char* conText = (con && con[closestConn]) ? con[closestConn] : "";
438 const ImU32* slotColorSource = i ? nodeTemplate.mOutputColors : nodeTemplate.mInputColors;
439 const ImU32 slotColor = slotColorSource ? slotColorSource[closestConn] : options.mDefaultSlotColor;
440 hoverSlot = true;
441 drawList->AddCircleFilled(closestPos, options.mNodeSlotRadius * options.mNodeSlotHoverFactor * 0.75f, IM_COL32(0, 0, 0, 200));
442 drawList->AddCircleFilled(closestPos, options.mNodeSlotRadius * options.mNodeSlotHoverFactor, slotColor);
443 drawList->AddText(io.FontDefault, 16, closestTextPos + ImVec2(1, 1), IM_COL32(0, 0, 0, 255), conText);
444 drawList->AddText(io.FontDefault, 16, closestTextPos, IM_COL32(250, 250, 250, 255), conText);
445 bool inputToOutput = (!editingInput && !i) || (editingInput && i);
446 if (nodeOperation == NO_EditingLink && !io.MouseDown[0] && !bDrawOnly)
447 {
448 if (inputToOutput)
449 {
450 // check loopback
451 Link nl;
452 if (editingInput)
453 nl = Link{nodeIndex, closestConn, editingNodeIndex, editingSlotIndex};
454 else
455 nl = Link{editingNodeIndex, editingSlotIndex, nodeIndex, closestConn};
456
457 if (!delegate.AllowedLink(nl.mOutputNodeIndex, nl.mInputNodeIndex))
458 {
459 break;
460 }
461 bool alreadyExisting = false;
462 for (size_t linkIndex = 0; linkIndex < linkCount; linkIndex++)
463 {
464 const auto link = delegate.GetLink(linkIndex);
465 if (!memcmp(&link, &nl, sizeof(Link)))
466 {
467 alreadyExisting = true;
468 break;
469 }
470 }
471
472 if (!alreadyExisting)
473 {
474 for (unsigned int linkIndex = 0; linkIndex < linkCount; linkIndex++)
475 {
476 const auto link = delegate.GetLink(linkIndex);
477 if (link.mOutputNodeIndex == nl.mOutputNodeIndex && link.mOutputSlotIndex == nl.mOutputSlotIndex)
478 {
479 delegate.DelLink(linkIndex);
480
481 break;
482 }
483 }
484
485 delegate.AddLink(nl.mInputNodeIndex, nl.mInputSlotIndex, nl.mOutputNodeIndex, nl.mOutputSlotIndex);
486 }
487 }
488 }
489 // when ImGui::IsWindowHovered() && !ImGui::IsAnyItemActive() is uncommented, one can't click the node
490 // input/output when mouse is over the node itself.
491 if (nodeOperation == NO_None &&
492 /*ImGui::IsWindowHovered() && !ImGui::IsAnyItemActive() &&*/ io.MouseClicked[0] && !bDrawOnly)
493 {
494 nodeOperation = NO_EditingLink;
495 editingInput = i == 0;
496 editingNodeSource = closestPos;
497 editingNodeIndex = nodeIndex;
498 editingSlotIndex = closestConn;
499 if (editingInput)
500 {
501 // remove existing link
502 for (unsigned int linkIndex = 0; linkIndex < linkCount; linkIndex++)
503 {
504 const auto link = delegate.GetLink(linkIndex);
505 if (link.mOutputNodeIndex == nodeIndex && link.mOutputSlotIndex == closestConn)
506 {
507 delegate.DelLink(linkIndex);
508 break;
509 }
510 }
511 }
512 }
513 }
514 }
515 return hoverSlot;
516}

519{

529 {
530 bool tenth = !(divy % 10);
531 drawList->AddLine(ImVec2(0.0f, y) + windowPos, ImVec2(canvasSize.x, y) + windowPos, tenth ? gridColor2 : gridColor);
532 }

720bool DrawMiniMap(ImDrawList* drawList, Delegate& delegate, ViewState& viewState, const Options& options, const ImVec2 windowPos, const ImVec2 canvasSize)
721{
722 if (Distance(options.mMinimap.Min, options.mMinimap.Max) <= FLT_EPSILON)
723 {
724 return false;
725 }
726
727 const size_t nodeCount = delegate.GetNodeCount();
728
729 if (!nodeCount)
730 {
731 return false;

733
734 ImVec2 min(FLT_MAX, FLT_MAX);
735 ImVec2 max(-FLT_MAX, -FLT_MAX);
736 const ImVec2 margin(50, 50);
737 for (NodeIndex nodeIndex = 0; nodeIndex < nodeCount; nodeIndex++)
738 {
739 const Node& node = delegate.GetNode(nodeIndex);
740 min = ImMin(min, node.mRect.Min - margin);
741 min = ImMin(min, node.mRect.Max + margin);
742 max = ImMax(max, node.mRect.Min - margin);
743 max = ImMax(max, node.mRect.Max + margin);

745
746 // add view in world space
747 const ImVec2 worldSizeView = canvasSize / viewState.mFactor;
748 const ImVec2 viewMin(-viewState.mPosition.x, -viewState.mPosition.y);
749 const ImVec2 viewMax = viewMin + worldSizeView;
750 min = ImMin(min, viewMin);
751 max = ImMax(max, viewMax);
752 const ImVec2 nodesSize = max - min;
753 const ImVec2 middleWorld = (min + max) * 0.5f;
754 const ImVec2 minScreen = windowPos + options.mMinimap.Min * canvasSize;
755 const ImVec2 maxScreen = windowPos + options.mMinimap.Max * canvasSize;
756 const ImVec2 viewSize = maxScreen - minScreen;
757 const ImVec2 middleScreen = (minScreen + maxScreen) * 0.5f;
758 const float ratioY = viewSize.y / nodesSize.y;
759 const float ratioX = viewSize.x / nodesSize.x;
760 const float factor = ImMin(ImMin(ratioY, ratioX), 1.f);
761
762 drawList->AddRectFilled(minScreen, maxScreen, IM_COL32(30, 30, 30, 200), 3, ImDrawFlags_RoundCornersAll);
763
764 for (NodeIndex nodeIndex = 0; nodeIndex < nodeCount; nodeIndex++)
765 {
766 const Node& node = delegate.GetNode(nodeIndex);
767 const auto nodeTemplate = delegate.GetTemplate(node.mTemplateIndex);
768
769 ImRect rect = node.mRect;
770 rect.Min -= middleWorld;
771 rect.Min *= factor;
772 rect.Min += middleScreen;
773
774 rect.Max -= middleWorld;
775 rect.Max *= factor;
776 rect.Max += middleScreen;
777
778 drawList->AddRectFilled(rect.Min, rect.Max, nodeTemplate.mBackgroundColor, 1, ImDrawFlags_RoundCornersAll);
779 if (node.mSelected)
780 {
781 drawList->AddRect(rect.Min, rect.Max, options.mSelectedNodeBorderColor, 1, ImDrawFlags_RoundCornersAll);
782 }
783 }
784
785 // add view
786 ImVec2 viewMinScreen = (viewMin - middleWorld) * factor + middleScreen;
787 ImVec2 viewMaxScreen = (viewMax - middleWorld) * factor + middleScreen;
788 drawList->AddRectFilled(viewMinScreen, viewMaxScreen, IM_COL32(255, 255, 255, 32), 1, ImDrawFlags_RoundCornersAll);
789 drawList->AddRect(viewMinScreen, viewMaxScreen, IM_COL32(255, 255, 255, 128), 1, ImDrawFlags_RoundCornersAll);

791 ImGuiIO& io = ImGui::GetIO();
792 const bool mouseInMinimap = ImRect(minScreen, maxScreen).Contains(io.MousePos);
793 if (mouseInMinimap && io.MouseClicked[0])
794 {
795 const ImVec2 clickedRatio = (io.MousePos - minScreen) / viewSize;
796 const ImVec2 worldPosCenter = ImVec2(ImLerp(min.x, max.x, clickedRatio.x), ImLerp(min.y, max.y, clickedRatio.y));
797
798 ImVec2 worldPosViewMin = worldPosCenter - worldSizeView * 0.5;
799 ImVec2 worldPosViewMax = worldPosCenter + worldSizeView * 0.5;
800 if (worldPosViewMin.x < min.x)
801 {
802 worldPosViewMin.x = min.x;
803 worldPosViewMax.x = worldPosViewMin.x + worldSizeView.x;
804 }
805 if (worldPosViewMin.y < min.y)
806 {
807 worldPosViewMin.y = min.y;
808 worldPosViewMax.y = worldPosViewMin.y + worldSizeView.y;
809 }
810 if (worldPosViewMax.x > max.x)
811 {
812 worldPosViewMax.x = max.x;
813 worldPosViewMin.x = worldPosViewMax.x - worldSizeView.x;
814 }
815 if (worldPosViewMax.y > max.y)
816 {
817 worldPosViewMax.y = max.y;
818 worldPosViewMin.y = worldPosViewMax.y - worldSizeView.y;
819 }
820 viewState.mPosition = ImVec2(-worldPosViewMin.x, -worldPosViewMin.y);
821 }
822 return mouseInMinimap;
