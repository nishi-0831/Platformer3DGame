254
255 while (!queue.empty())
256 {

258 auto& func = queue.front().second;

270 if (ImGui::Selectable(name.c_str(), selectedName == name))
271 {
272 isSelected = true;
273 imguiWindowStates_[_show].selectedName = name;
274 selectedFunc = func;
275 }
276
277 queue.pop();
278 }
279 ImGui::EndChild();

281 // Listの横に property表示

453void mtgb::MTImGui::DrawCone(const Vector3& _position, const Vector3& _direction, float _fovAngleDegree, float _distance, float _thickness, int _segments)
454{
455 // 正規化された方向ベクトル
456 Vector3 forward = Vector3::Normalize(_direction);
457
458 // 円錐の底面の中心
459 Vector3 baseCenter = _position + forward * _distance;
460
461 // 円錐の底面の半径
462 // MEMO: tanθ = 対辺 / 底辺 : 対辺 = 底辺 * tanθ
463 float baseRadius = _distance * std::tanf(DirectX::XMConvertToRadians(_fovAngleDegree / 2.0f));
464
465 // 上下方向を決定
466 Vector3 up = std::fabsf(DirectX::XMVectorGetY(forward)) < 0.9f
467 ? Vector3{ 0.0f,1.0f,0.0f }
468 : Vector3{ 1.0f,0.0f,0.0f };
469
470 // 右方向を計算
471 Vector3 right = Vector3::Normalize(Vector3::Cross(forward, up));
472
473 // 上方向を再計算
474 up = Vector3::Cross(right, forward);
475
476 // 底面の円周上の点を計算
477 std::vector<Vector3> baseCirclePoints(_segments);
478 for (int i = 0; i < _segments; i++)
479 {
480 float angle = DirectX::XM_2PI * i / _segments;
481 float x = baseRadius * std::cosf(angle);
482 float y = baseRadius * std::sinf(angle);
483 baseCirclePoints[i] = baseCenter + right * x + up * y;
484 }
485
486 // 円錐の側面を描画
487 for (int i = 0; i < _segments; i++)
488 {
489 DrawLine(_position, baseCirclePoints[i], _thickness);
490 }
491
492 // 底面の円を描画
493 for (int i = 0; i < _segments; i++)
494 {
495 int nextIdx = (i + 1) % _segments;
496 DrawLine(baseCirclePoints[i], baseCirclePoints[nextIdx], _thickness);
497 }
498}
