50void mtgb::AudioClip::LoadWave(mtbin::MemoryStream& _ms, const byte* _first4)
51{
52 // チャンク識別子は 4 byte
53 static const size_t ID_SIZE{ 4 };
54
55 // チャンクヘッダ情報
56 struct ChunkHeader
57 {
58 ChunkHeader() :
59 id{},
60 size{ 0 }
61 {}
62
63 byte id[ID_SIZE]; // チャンク識別子
64 uint32_t size; // チャンクボディのサイズ
65 };
66
67 // RIFF チャンク
68 ChunkHeader riff{};
69 _ms.Read(riff.id, sizeof(riff.id), ID_SIZE);
70 massert(CompareId<ID_SIZE>(riff.id, "RIFF")
71 && "RIFF チャンクIdの不一致 @AudioClip::Load");
72 riff.size = _ms.Read<uint32_t>();
73
74 // WAVE チャンク
75 byte wave[ID_SIZE]{};
76 _ms.Read(wave, sizeof(wave), ID_SIZE);
77 massert(CompareId<ID_SIZE>(wave, "WAVE")
78 && "WAVE チャンクIdの不一致 @AudioClip::Load");
79
80 // フォーマットチャンク
81 ChunkHeader format{};
82 _ms.Read(format.id, sizeof(format.id), ID_SIZE);
83 massert(CompareId<ID_SIZE>(format.id, "fmt ")
84 && "フォーマット チャンクIdの不一致 @AudioClip::Load");
85 format.size = _ms.Read<uint32_t>();
86 massert((format.size == 0x10)
87 && "対応していないフォーマットサイズ @AudioClip::Load");
88 _ms.Read(reinterpret_cast<byte*>(&pWaveData_->waveFormat), sizeof(WAVEFORMATEX), format.size);
89
90 // wBitsPerSampleを設定
91 /*pWaveData_->waveFormat.wBitsPerSample =
92 pWaveData_->waveFormat.nBlockAlign * 8 / pWaveData_->waveFormat.nChannels;*/
93 pWaveData_->waveFormat.nBlockAlign =
94 pWaveData_->waveFormat.nChannels * pWaveData_->waveFormat.wBitsPerSample / 8;
95 pWaveData_->waveFormat.nAvgBytesPerSec = pWaveData_->waveFormat.nSamplesPerSec * pWaveData_->waveFormat.nBlockAlign;
96
97 // dataチャンクを探す
98 ChunkHeader header{};
99 while (true)
100 {
101 _ms.Read(header.id, sizeof(header.id), ID_SIZE);
102 header.size = _ms.Read<uint32_t>();
103
104 if (CompareId<ID_SIZE>(header.id, "data"))
105 {
106 break; // dataチャンクに到達したら離脱
107 }
108 else
109 {
110 _ms.Seek(_ms.Current() + header.size); // dataチャンクでないなら読み飛ばし
111 }
112 }
113
114 pWaveData_->bufferSize = header.size;
115 pWaveData_->pBuffer = new byte[header.size]{};
116 _ms.Read(pWaveData_->pBuffer, header.size, header.size);
117}

119void mtgb::AudioClip::LoadMp3(mtbin::MemoryStream& _ms, const byte* _first4)
120{
121 // MemoryStreamからmp3データを読み込む
122 size_t size = _ms.GetLength();
123 byte* mp3_data = new byte[size];
124 _ms.Read(mp3_data, static_cast<int>(size), static_cast<int>(size));
125
126 // drmp3のデコード初期化
127 drmp3 config;
128 if(!drmp3_init_memory(&config, mp3_data, size, NULL))
129 {
130 massert(false && "MP3の初期化に失敗 @AudioClip::LoadMp3");
131 delete[] mp3_data;
132 }
133
134 // 総PCMサンプル数を取得
135 drmp3_uint64 total_samples = drmp3_get_pcm_frame_count(&config);
136 // バッファサイズを計算
137 size_t pcm_sample_count = (size_t)(total_samples * config.channels);
138 size_t pcm_bytes = pcm_sample_count * sizeof(short);
139
140 // PCM出力バッファ確保
141 pWaveData_->bufferSize = pcm_bytes;
142 pWaveData_->pBuffer = new byte[pcm_bytes];
143
144 // PCMデータを読み込む
145 drmp3_read_pcm_frames_s16(&config, total_samples, reinterpret_cast<short*>(pWaveData_->pBuffer));
146
147 // WAVEフォーマット情報設定
148 WAVEFORMATEX& wf = pWaveData_->waveFormat;
149 wf.wFormatTag = 1;
150 wf.nChannels = static_cast<WORD>(config.channels);
151 wf.nSamplesPerSec = static_cast<DWORD>(config.sampleRate);
152 wf.wBitsPerSample = 16;
153 wf.nBlockAlign = wf.nChannels * wf.wBitsPerSample / 8;
154 wf.nAvgBytesPerSec = wf.nSamplesPerSec * wf.nBlockAlign;
155
156 // 後始末
157 drmp3_uninit(&config);
158 delete[] mp3_data; // メモリリーク防止のために開放
159}
